88  const external_flash_device* possible_device = &possible_devices[i];
89  if (jedec_ids[0] == possible_device->manufacturer_id &&
90  jedec_ids[1] == possible_device->memory_type &&

92  _flash_dev = possible_device;
93  break;
94  }
95  }
96 
97  if (_flash_dev == NULL) return false;
98 
99  // We don't know what state the flash is in so wait for any remaining writes and then reset.
100 
101  // The write in progress bit should be low.
102  while ( readStatus() & 0x01 ) {}
103 
104  // The suspended write/erase bit should be low.
105  while ( readStatus2() & 0x80 ) {}
106 
107  QSPI0.runCommand(QSPI_CMD_ENABLE_RESET);
108  QSPI0.runCommand(QSPI_CMD_RESET);
109 
110  // Wait 30us for the reset
111  delayMicroseconds(30);
112 
113  // Speed up to max device frequency
114  QSPI0.setClockSpeed(_flash_dev->max_clock_speed_mhz*1000000UL);
115 
116  // Enable Quad Mode if available
117  if (_flash_dev->quad_enable_bit_mask)
118  {
119  // Verify that QSPI mode is enabled.
120  uint8_t status = _flash_dev->single_status_byte ? readStatus() : readStatus2();
121 
122  // Check the quad enable bit.
123  if ((status & _flash_dev->quad_enable_bit_mask) == 0)
124  {

133  } else {

135  }
136  }

261 uint32_t Adafruit_QSPI_Flash::writeBuffer (uint32_t addr, uint8_t *data, uint32_t len)
262 {
263  if (!_flash_dev) return 0;
264 
265  uint32_t remain = len;
266 
267  //write one page at a time
268  while(remain)
269  {
270  _wait_for_flash_ready();
271  writeEnable();
272 
273  uint16_t const toWrite = min(remain, QSPI_FLASH_PAGE_SIZE);
274 
275  if ( !QSPI0.writeMemory(addr, data, toWrite) ) break;
276 
277  remain -= toWrite;
278  data += toWrite;
279  addr += toWrite;
280  }
281 
282  return len - remain;
283 }
